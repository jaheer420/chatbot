from fastapi import FastAPI
import mysql.connector
import requests

app = FastAPI()


# -------------------------
# DATABASE CONNECTION
# -------------------------
def get_all_courses():
    conn = mysql.connector.connect(
        host="localhost",
        user="root",
        password="",
        database="innovel_db"
    )

    cursor = conn.cursor(dictionary=True, buffered=True)

    cursor.execute("SELECT * FROM courses")
    courses = cursor.fetchall()

    cursor.close()
    conn.close()

    return courses


# -------------------------
# EXISTING COURSE ENDPOINT (UNCHANGED)
# -------------------------
@app.get("/course/{name}")
def course_details(name: str):

    conn = mysql.connector.connect(
        host="localhost",
        user="root",
        password="",
        database="innovel_db"
    )

    cursor = conn.cursor(dictionary=True, buffered=True)

    cursor.execute(
        "SELECT * FROM courses WHERE course_name LIKE %s",
        (f"%{name}%",)
    )

    course = cursor.fetchone()

    cursor.close()
    conn.close()

    if not course:
        return {"message": "Course not found"}

    final_fee = course["offer_fees"] if course["offer_fees"] else course["fees"]

    return {
        "course": course["course_name"],
        "duration": f"{course['duration_hours']} hours",
        "fees": final_fee,
        "syllabus": course["syllabus_highlight"]
    }


# -------------------------
# OLLAMA FUNCTION
# -------------------------
def ask_ollama(prompt):
    response = requests.post(
        "http://localhost:11434/api/generate",
        json={
            "model": "llama3",
            "prompt": prompt,
            "stream": False
        }
    )

    return response.json()["response"]


# -------------------------
# FULLY AUTOMATIC INNOVEL AI
# -------------------------
@app.get("/chat/{message}")
def innovel_ai(message: str):

    courses = get_all_courses()
    message_lower = message.lower()

    matched_course = None

    # 🔎 Automatically detect course from DB (NO HARDCODING)
    for course in courses:
        if course["course_name"].lower() in message_lower:
            matched_course = course
            break

    # ✅ If course detected → inject only that course
    if matched_course:

        final_fee = matched_course["offer_fees"] if matched_course["offer_fees"] else matched_course["fees"]

        course_context = f"""
Course Name: {matched_course['course_name']}
Duration: {matched_course['duration_hours']} hours
Fees: ₹{final_fee}
Highlights: {matched_course['syllabus_highlight']}
"""

    else:
        # 🔁 If no course detected → inject all courses
        course_context = ""

        for course in courses:
            final_fee = course["offer_fees"] if course["offer_fees"] else course["fees"]

            course_context += f"""
Course Name: {course['course_name']}
Duration: {course['duration_hours']} hours
Fees: ₹{final_fee}
Highlights: {course['syllabus_highlight']}
------------------------
"""

    prompt = f"""
You are Innovel Institute Official AI Assistant.

Use ONLY the provided course data to answer course-related questions.
If user asks about fees, duration, syllabus, recommend courses etc,
you MUST answer from the data below.

Do NOT say fees vary.
Do NOT guess.
Be confident and professional.

Course Data:
{course_context}

User Question:
{message}

Give structured and clear answer.
"""

    ai_reply = ask_ollama(prompt)

    return {"reply": ai_reply}
